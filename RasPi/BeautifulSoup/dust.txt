function doExtExcelDownload(url, fileName, headerTitle, gridData, searchFieldId) {
	var f = document.getElementById('_excelForm');

	var orgTarget = f.target;
	var orgAction = f.action;
	window.open('','excel_Hidden','top=0, left=0, width=520, height=500, toolbar=0, directories=0, status=0, menubar=0, scrollbars=yes, resizable=yes');

	f.target = "excel_Hidden";
    f.action = "/"+url;

	f.excelFileName.value = fileName;
	f.excelGridData.value = gridData;
	f.excelHeaderTitle.value = headerTitle;

	if(searchFieldId != undefined) {
		var searchValue = "";
		for(var i=0; i < searchFieldId.length; i++) {
			if(typeof(document.getElementsByName(searchFieldId[i])) != 'undefined'){
				searchValue += document.getElementsByName(searchFieldId[i])[0].value + "\t";
			}
		}
		f.searchValue.value = searchValue;
	}

	f.process = 'false'; //엑셀다운로드시엔 프로그래스바 안띄움 (닫히는 시점을 알수없음)
	f.spiderSubmit();

	f.target = orgTarget;
	f.action = orgAction;

	submitStateFalse(f);
}

						elementObj.type.toString().toLowerCase() == "radio") {
